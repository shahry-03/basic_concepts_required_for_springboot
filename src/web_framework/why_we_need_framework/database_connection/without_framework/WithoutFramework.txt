Plain Java me tumhe sab kuch khud karna padta hai. Connection banana, error sambhalna, aur sabse zarooriâ€”Connection close karna (jo aksar log bhool jate hain aur server crash ho jata hai).

Is code ko "Boilerplate Code" kehte hain (faltu ka code jo baar baar likhna padta hai).
import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.SQLException;

public class UserManager {

    public void saveUser(String name, String email) {
        // Variables declare karne padenge taake finally block me close kar sakein
        Connection connection = null;
        PreparedStatement statement = null;

        try {
            // 1. Driver Load karo
            Class.forName("com.mysql.cj.jdbc.Driver");

            // 2. Connection Open karo (Ye har baar likhna padega)
            connection = DriverManager.getConnection(
                    "jdbc:mysql://localhost:3306/mydb", "root", "password");

            // 3. SQL Query likho
            String sql = "INSERT INTO users (name, email) VALUES (?, ?)";

            // 4. Statement banao
            statement = connection.prepareStatement(sql);
            statement.setString(1, name);
            statement.setString(2, email);

            // 5. Execute karo
            statement.executeUpdate();
            System.out.println("User saved!");

        } catch (Exception e) {
            // 6. Errors handle karo
            e.printStackTrace();
        } finally {
            // 7. CONNECTION CLOSE KARNA (Sabse important aur boring kaam)
            // Agar ye bhool gaye, to database server full ho jayega
            try {
                if (statement != null) statement.close();
                if (connection != null) connection.close();
            } catch (SQLException e) {
                e.printStackTrace();
            }
        }
    }
}
